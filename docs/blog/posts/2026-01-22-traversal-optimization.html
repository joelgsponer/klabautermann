Chart the Course

How many hops from John to Acme? Who blocks this task? What's connected within two degrees?

Questions that need answers. Fast.

Traversal Configuration

Control how you navigate the graph:

from klabautermann.memory import (
    TraversalConfig,
    TraversalDirection,
)

config = TraversalConfig(
    max_depth=3,              # How deep to go
    direction=TraversalDirection.BOTH,  # outgoing, incoming, or both
    relationship_types=None,  # Filter by relationship (or None for all)
    include_expired=False,    # Skip expired relationships
    limit=100,                # Cap results
)

Traverse from a Node

Start at one node, explore outward:

from klabautermann.memory import traverse_from_node
from klabautermann.core.ontology import NodeLabel

result = await traverse_from_node(
    client=neo4j_client,
    start_uuid="john-uuid",
    start_label=NodeLabel.PERSON,
    config=config,
)

print(f"Found {len(result.nodes)} connected nodes")
print(f"Query took {result.stats.query_time_ms:.1f}ms")

Find Shortest Path

Two nodes. Find the connection:

from klabautermann.memory import find_shortest_path

result = await find_shortest_path(
    client=neo4j_client,
    from_uuid="john-uuid",
    to_uuid="acme-uuid",
    max_depth=6,
)

if not result.is_empty:
    print(f"Path length: {result.stats.depth_reached}")
    for node in result.nodes:
        print(f"  -> {node['name']}")

Task Dependency Chains

What blocks what?

from klabautermann.memory import traverse_dependency_chain

# Find everything blocking this task
blockers = await traverse_dependency_chain(
    client=neo4j_client,
    task_uuid="task-uuid",
    direction=TraversalDirection.INCOMING,
)

for task in blockers.nodes:
    print(f"Blocked by: {task['name']} ({task['status']})")

Connected Neighborhoods

What's nearby?

from klabautermann.memory import find_connected_entities

# Find everything within 2 hops
neighbors = await find_connected_entities(
    client=neo4j_client,
    entity_uuid="person-uuid",
    hops=2,  # Capped at 3 for performance
)

for entity in neighbors.nodes:
    print(f"{entity['type']}: {entity['name']} ({entity['distance']} hops)")

Performance Benchmarking

Measure before optimizing:

from klabautermann.memory import benchmark_traversal

result = await benchmark_traversal(
    client=neo4j_client,
    operation_name="person_network",
    query="MATCH (p:Person)-[r*1..3]-(n) WHERE p.uuid = $uuid RETURN n",
    params={"uuid": "test-uuid"},
    iterations=10,
)

print(f"Average: {result.avg_time_ms:.2f}ms")
print(f"Min: {result.min_time_ms:.2f}ms")
print(f"Max: {result.max_time_ms:.2f}ms")

Index Hints

Know which indexes exist:

from klabautermann.memory import (
    RELATIONSHIP_INDEXES,
    NODE_SEARCH_INDEXES,
    get_index_hint,
    get_search_index,
)

# Relationship indexes for temporal queries
print(RELATIONSHIP_INDEXES)
# {WORKS_AT: "works_at_temporal", LOCATED_IN: "located_in_temporal", ...}

# Node fulltext search indexes
print(NODE_SEARCH_INDEXES)
# {PERSON: "person_search", ORGANIZATION: "org_search", ...}

Traversal Stats

Every traversal returns statistics:

stats = result.stats

print(f"Query time: {stats.query_time_ms:.1f}ms")
print(f"Nodes visited: {stats.nodes_visited}")
print(f"Relationships traversed: {stats.relationships_traversed}")
print(f"Paths found: {stats.paths_found}")
print(f"Max depth reached: {stats.depth_reached}")
