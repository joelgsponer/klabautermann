The Thread Model

Gmail's threads API returns the complete conversation - every message, every participant, every timestamp. The model captures this richness while adding computed properties for quick reference:

class EmailThread(BaseModel):
    """A complete email thread with all messages."""

    id: str  # Thread ID
    subject: str
    messages: list[EmailMessage]
    participant_count: int = 0
    message_count: int = 0

    @property
    def participants(self) -> list[str]:
        """Get unique participants in the thread."""
        senders = {msg.sender for msg in self.messages}
        recipients = {msg.recipient for msg in self.messages if msg.recipient}
        return list(senders | recipients)

    @property
    def date_range(self) -> str:
        """Human-readable date range of the thread."""
        if not self.messages:
            return "No messages"
        dates = sorted(msg.date for msg in self.messages)
        start = dates[0].strftime("%b %d")
        end = dates[-1].strftime("%b %d, %Y")
        if start == end.split(",")[0]:
            return end
        return f"{start} - {end}"

The participants property deduplicates across all messages. The date_range property handles the common case of same-day threads gracefully - no need for "Jan 22 - Jan 22, 2026".

Fetching the Thread

Gmail's threads.get returns all messages in one call. The parsing follows the same pattern as individual message fetching:

async def get_thread(
    self,
    thread_id: str,
    context: Any = None,
) -> EmailThread | None:
    """Get a complete email thread with all messages."""
    await self.start()

    def do_get_thread() -> dict[str, Any]:
        return (
            self._gmail_service.users()
            .threads()
            .get(userId="me", id=thread_id, format="full")
            .execute()
        )

    try:
        raw_thread = await self._rate_limited_call("gmail", do_get_thread)
    except Exception:
        return None

    messages: list[EmailMessage] = []
    for raw_msg in raw_thread.get("messages", []):
        # Parse each message using existing email parsing logic
        messages.append(self._parse_email_message(raw_msg))

    return EmailThread(
        id=thread_id,
        subject=messages[0].subject if messages else "No Subject",
        messages=messages,
        participant_count=len(set(m.sender for m in messages)),
        message_count=len(messages),
    )

The format="full" parameter ensures we get complete message bodies, not just headers. Essential for meaningful summarization.

The Summary Model

A good summary is more than a condensed version of the text. It identifies what matters:

class EmailThreadSummary(BaseModel):
    """Summary of an email thread."""

    thread_id: str
    subject: str
    summary: str  # 2-3 sentence summary of the conversation
    key_points: list[str] = Field(default_factory=list)
    action_items: list[str] = Field(default_factory=list)
    participants: list[str] = Field(default_factory=list)
    message_count: int = 0
    date_range: str = ""
    sentiment: str = "neutral"  # positive, negative, neutral

The action_items field is particularly useful. Buried in message 23 of a 47-message thread is often the one sentence that actually requires you to do something. Now it surfaces to the top.

LLM-Powered Summarization

The summarization uses Claude Haiku with the tool_use pattern for structured output:

async def summarize_email_thread(
    self,
    thread_id: str,
    context: Any = None,
) -> EmailThreadSummary | None:
    """Generate an AI-powered summary of an email thread."""
    thread = await self.get_thread(thread_id, context)
    if not thread or not thread.messages:
        return None

    # Format thread for LLM
    formatted_thread = self._format_thread_for_summary(thread)

    # Use Claude Haiku for summarization
    client = anthropic.Anthropic()

    response = client.messages.create(
        model="claude-3-5-haiku-20241022",
        max_tokens=1024,
        tools=[{
            "name": "create_summary",
            "description": "Create a structured summary of an email thread",
            "input_schema": {
                "type": "object",
                "properties": {
                    "summary": {"type": "string"},
                    "key_points": {"type": "array", "items": {"type": "string"}},
                    "action_items": {"type": "array", "items": {"type": "string"}},
                    "sentiment": {"type": "string", "enum": ["positive", "negative", "neutral"]},
                },
                "required": ["summary", "key_points", "action_items", "sentiment"],
            },
        }],
        tool_choice={"type": "tool", "name": "create_summary"},
        messages=[{
            "role": "user",
            "content": f"Summarize this email thread:\n\n{formatted_thread}",
        }],
    )

The tool_choice parameter forces Claude to respond with structured JSON matching our schema. No parsing ambiguity, no "here's the summary:" preambles to strip out.

Formatting for the LLM

The format matters. Too verbose and you waste tokens. Too sparse and the LLM misses context:

def _format_thread_for_summary(self, thread: EmailThread) -> str:
    """Format a thread for LLM summarization."""
    lines = [f"Subject: {thread.subject}", ""]

    for i, msg in enumerate(thread.messages, 1):
        lines.append(f"Message {i}:")
        lines.append(f"From: {msg.sender}")
        lines.append(f"Date: {msg.date.strftime('%Y-%m-%d %H:%M')}")
        if msg.body:
            # Truncate very long messages
            body = msg.body[:2000] + "..." if len(msg.body) > 2000 else msg.body
            lines.append(f"Content: {body}")
        lines.append("")

    return "\n".join(lines)

The 2000-character truncation per message keeps the context window manageable for threads with massive individual messages while preserving the conversational flow.
